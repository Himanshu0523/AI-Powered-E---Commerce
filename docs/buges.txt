useEffect is a hook used to handle side effects in a component.
    // Run once
    useEffect(() => {}, []);

    // Run when value changes
    useEffect(() => {}, [value]);

    // Cleanup (important)
    useEffect(() => {
        return () => {
            console.log("Component unmounted");
        };
    }, []);
